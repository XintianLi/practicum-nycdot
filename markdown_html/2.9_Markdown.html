#trips by days
ggplot()+
  geom_sf(data = rideApr) +
  facet_wrap(~days)

#trips by hours in one day
ggplot()+
  geom_sf(data = rideApr %>%
            filter(days == "01")) +
  facet_wrap(~rhour)

#change days as numeric
rideApr$days <- as.numeric(rideApr$days)

#trips over days
ggplot(rideApr %>%
       as.data.frame() %>%
       group_by(days) %>%
       tally())+
  geom_line(aes(x = days, y = n))

#trips over hour
ggplot(rideApr %>%
         as.data.frame() %>%
         group_by(rhour) %>%
         tally())+
  geom_line(aes(x = rhour, y = n))
